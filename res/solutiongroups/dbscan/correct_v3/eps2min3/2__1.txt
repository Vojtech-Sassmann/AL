def fibonacci(n):
    fib=[1,1]
    for i in range(2,n):
        fib.append(fib[i-2]+fib[i-1])
    for i in range(n):
        print(fib[i])
    

def fibonacci(n):
    fib = [1, 1]
    for i in range(n):
        fib.append(fib[len(fib)-1]+fib[len(fib)-2])
    for i in range(n):
    	print(fib[i])
