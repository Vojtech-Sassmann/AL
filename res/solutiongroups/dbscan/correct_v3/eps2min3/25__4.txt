def count_a(text):
    cnt=0
    text=text.lower()
    for let in text:
        if let=='a':
            cnt+=1
    return(cnt)

    

def count_a(text):
    text = text.upper()
    countedas = 0
    for i in range(len(text)):
        if text[i] == 'A': 
            countedas += 1
    return countedas
