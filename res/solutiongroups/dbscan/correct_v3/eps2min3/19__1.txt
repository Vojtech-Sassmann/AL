def zigzag(text):
    for i in range(len(text)):
        if i%2 == 0:
            print(text[i], end="")
        else:
            print(".", end="")
    print()
    for j in range(len(text)):
        if j%2 == 1:
            print(text[j], end="")
        else:
            print(".", end="")
