def count_a(text):
    cnt=0
    text=text.lower()
    for let in text:
        if let=='a':
            cnt+=1
    return(cnt)

    

def count_a(text):
    sum_a = 0
    analysis = text.lower()
    for character in analysis:
        if character == "a":
            sum_a+=1
    return sum_a
