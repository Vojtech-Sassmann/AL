def zigzag(text):
    l1 = []
    l2 = []
    for i, x in enumerate(text):
        if i % 2 == 0:
            l1.append(x)
            l2.append(".")
        else:
            l1.append(".")
            l2.append(x)
    print("".join(l1))
    print("".join(l2))
