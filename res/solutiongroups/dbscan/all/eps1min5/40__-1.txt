def bigN(n):
    for rows in range(1,n+1):
        for cols in range(1,n+3):
            if cols-rows==1:
                print("\\",end="")
            elif cols==1 or cols==n+2:
                print("|",end="")
            else:
                print(" ",end="")
        print()

def bigN(n):
    prvni_retezec = ""
    druhy_retezec = " "*(n-1)
    for i in range(n):
        retezec = "|{0}\\{1}|".format(prvni_retezec, druhy_retezec)
        prvni_retezec=prvni_retezec+" "
        druhy_retezec=druhy_retezec[:-1]
        print(retezec)

def middle(n,pos):
    for i in range(n):
    	if pos != i:
        	print(" ",end="")
    	else:
        	print("\\",end="")
def bigN(n):
    for i in range(n):
        print("|", end="")
        middle(n,i)
        print("|")

def bigN(n):
    for row in range(0, n):
        print("|", end="")
        print(" "*row, end="")
        print("\\", end="")
        print(" "*((n-1)-row), end="")
        print("|", end="")
        print()

def bigN(n):
    for i in range(n):
