def greatest_common_divisor(a, b):
    s=[]
    t=[]
    for i in range (a):
        if a%(i+1)==0: s.append(i+1)
    for j in range (b):
        if b%(j+1)==0: s.append(j+1)
    for k in range (len(s)):
        for l in range (len(s)):
            if k>l and s[k]==s[l]:t.append(s[k])
    t.sort
    return(t[len(t)-1])

    while b:
        a, b = b, a%b
    return a
